class Solution:
    def spiralOrder(self, matrix: list[list[int]]) -> list[int]:
        if not matrix:
            return []
        
        # Initialize the four boundaries
        top, bottom = 0, len(matrix) - 1
        left, right = 0, len(matrix[0]) - 1
        res = []
        
        while left <= right and top <= bottom:
            # 1. Traverse from left to right along the top row
            for i in range(left, right + 1):
                res.append(matrix[top][i])
            top += 1  # Shrink the top boundary down
            
            # 2. Traverse from top to bottom along the right column
            for i in range(top, bottom + 1):
                res.append(matrix[i][right])
            right -= 1  # Shrink the right boundary left
            
            # 3. Traverse from right to left along the bottom row
            # Check condition to prevent repeating rows in non-square matrices
            if top <= bottom:
                for i in range(right, left - 1, -1):
                    res.append(matrix[bottom][i])
                bottom -= 1  # Shrink the bottom boundary up
                
            # 4. Traverse from bottom to top along the left column
            # Check condition to prevent repeating columns in non-square matrices
            if left <= right:
                for i in range(bottom, top - 1, -1):
                    res.append(matrix[i][left])
                left += 1  # Shrink the left boundary right
                
        return res
